import math

#degrees --> radians
print("Enter an angle value in degrees: ")

degr = input('> ')

degr = float(degr)

rads = math.radians(degr)

#trig functions!!

soh = math.sin(rads)
cah = math.cos(rads)
toa = math.tan(rads)

print("{:.8} in radians is {:.8} and the sine is: {:.8}, cosine is: {:.8}, tangent is: {:.8}" .format(degr, rads, soh, cah, toa))
